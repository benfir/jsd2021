  https://m01.cluster02.local:6443

  "paths": [
    "/api",
    "/api/v1",
    "/apis",
    "/apis/",
    ...
   "/version"
  ]
}
----

$ source ~/bin/prepare-env.source

$ export CRT=/home/bf/.kube/cluster02-kubeadmin.crt
$ export KEY=/home/bf/.kube/cluster02-kubeadmin.key
--
